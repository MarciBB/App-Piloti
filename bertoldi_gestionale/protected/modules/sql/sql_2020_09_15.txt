CREATE TABLE RT_PrenotazioneMessaggio (
  `PrenotazioneMessaggioId` INT(11) NOT NULL AUTO_INCREMENT,
  `PrenotazioneId` INT(11) NULL,
  `Messaggio` VARCHAR(255) NULL,
  `Numero` VARCHAR(45) NULL,
  `Canale` VARCHAR(45) NULL,
  `DataIns` DATETIME NULL,
  `OpeIns` INT(11) NULL,
  PRIMARY KEY (`PrenotazioneMessaggioId`));

  `GooglePeopleId` INT NOT NULL AUTO_INCREMENT,

  PRIMARY KEY (`GooglePeopleId`));

  `PrenotazioneNumeroId` INT(10) NULL DEFAULT NULL,

  `Ordine` INT(4) NULL DEFAULT NULL,

  `GiorniAggiuntivi` INT(1) NULL DEFAULT NULL,
